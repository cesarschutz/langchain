Argumentos de Funções - Posicionais vs Nomeados

CONCEITO FUNDAMENTAL

Este é o conceito que você perguntou! Em Python, existem várias formas de passar argumentos para funções, oferecendo muito mais flexibilidade que Java.

Exemplo do Projeto LangChain 
conversational_chain = RunnableWithMessageHistory(
    chain,                           # ← Argumento POSICIONAL (sem nome)
    get_session_history,             # ← Argumento POSICIONAL (sem nome)
    input_messages_key="input",     # ← Argumento NOMEADO (com nome=valor)
    history_messages_key="history",  # ← Argumento NOMEADO (com nome=valor)
)

Argumentos Posicionais

  • Passados SEM nome
  • Ordem IMPORTA
  • Geralmente obrigatórios
  • Vão primeiro na lista

Argumentos Nomeados (kwargs)

  • Passados COM nome=valor
  • Ordem NÃO importa
  • Geralmente opcionais
  • Vão depois dos posicionais

⚠️ REGRA DE OURO

Argumentos posicionais SEMPRE vêm ANTES dos nomeados!

Exemplo de Erro Comum 
# ✅ CORRETO:
funcao(arg1, arg2, nome="valor", outro="valor2")

# ❌ ERRADO:
funcao(nome="valor", arg1, arg2)  # SyntaxError!

Argumentos Posicionais em Detalhes

Criando Funções com Argumentos Posicionais 
# Função simples com argumentos posicionais
def criar_usuario(nome, idade, cidade):
    """Cria um usuário com informações básicas."""
    return {
        "nome": nome,
        "idade": idade,
        "cidade": cidade,
        "ativo": True
    }

# Chamadas corretas (ordem importa!)
usuario1 = criar_usuario("João", 25, "São Paulo")
usuario2 = criar_usuario("Maria", 30, "Rio de Janeiro")

# ❌ Erro: ordem errada
# usuario_erro = criar_usuario(25, "João", "São Paulo")  # Dados trocados!

print(usuario1)
# Output: {'nome': 'João', 'idade': 25, 'cidade': 'São Paulo', 'ativo': True}

💡 Quando usar argumentos posicionais?

  • Poucos argumentos (2-3 no máximo)
  • Ordem óbvia (nome, idade, email)
  • Argumentos obrigatórios
  • APIs simples e diretas

*args - Argumentos Posicionais Variáveis

Criando Funções que Recebem *args 
# *args captura argumentos posicionais extras como tupla
def somar_numeros(primeiro, *args):
    """Soma o primeiro número com todos os outros fornecidos."""
    total = primeiro
    print(f"Primeiro número: {primeiro}")
    print(f"Números extras: {args}")  # args é uma tupla!
    
    for numero in args:
        total += numero
    
    return total

# Testando com diferentes quantidades de argumentos
resultado1 = somar_numeros(10)                    # Só o obrigatório
resultado2 = somar_numeros(10, 5)                 # 1 extra
resultado3 = somar_numeros(10, 5, 3, 7, 2)       # Vários extras

print(f"Resultado 1: {resultado1}")  # 10
print(f"Resultado 2: {resultado2}")  # 15
print(f"Resultado 3: {resultado3}")  # 27

# Exemplo mais avançado: função de log
def log_message(level, *messages):
    """Sistema de log que aceita múltiplas mensagens."""
    timestamp = "2024-01-01 10:00:00"
    combined_message = " | ".join(messages)
    print(f"[{timestamp}] {level.upper()}: {combined_message}")

log_message("info", "Sistema iniciado")
log_message("error", "Falha na conexão", "Tentativa 3", "Timeout: 30s")

# **kwargs captura argumentos nomeados extras como dicionário
def configurar_servidor(**kwargs):
    """Configura um servidor com opções flexíveis."""
    # Valores padrão
    config = {
        "host": "localhost",
        "porta": 8080,
        "debug": False,
        "timeout": 30
    }
    
    # Atualiza com os valores fornecidos
    config.update(kwargs)
    
    print("Configuração do servidor:")
    for chave, valor in config.items():
        print(f"  {chave}: {valor}")
    
    return config

# Testando com diferentes configurações

Função com Todos os Tipos de Argumentos 
def funcao_completa(obrigatorio, *args, padrao="valor_padrao", **kwargs):
    """
    Demonstra todos os tipos de argumentos em uma função.
    
    Args:
        obrigatorio: Argumento obrigatório posicional
        *args: Argumentos posicionais extras
        padrao: Argumento com valor padrão
        **kwargs: Argumentos nomeados extras
    """
    print(f"Obrigatório: {obrigatorio}")
    print(f"Args extras: {args}")
    print(f"Padrão: {padrao}")
    print(f"Kwargs: {kwargs}")
    print("-" * 40)

# Testando todas as combinações
funcao_completa("A")
# Obrigatório: A
# Args extras: ()
# Padrão: valor_padrao
# Kwargs: {}

funcao_completa("A", "B", "C")
# Obrigatório: A
# Args extras: ('B', 'C')
# Padrão: valor_padrao
# Kwargs: {}

funcao_completa("A", "B", "C", padrao="custom")
# Obrigatório: A
# Args extras: ('B', 'C')
# Padrão: custom
# Kwargs: {}

funcao_completa("A", "B", padrao="custom", extra1="X", extra2="Y")
# Obrigatório: A
# Args extras: ('B',)
# Padrão: custom
# Kwargs: {'extra1': 'X', 'extra2': 'Y'}

Imports e Módulos

O que são Imports?

Imports permitem usar código de outros arquivos ou bibliotecas. É como "pegar emprestado" funcionalidades prontas, evitando reescrever código.

Por que usar imports?

  • Reutilização: Não reinvente a roda
  • Organização: Separe código em módulos
  • Manutenção: Mudanças em um lugar
  • Colaboração: Use bibliotecas da comunidade
Exemplos do Projeto LangChain 
# Imports típicos do projeto LangChain
from langchain_openai import ChatOpenAI          # Classe específica
from langchain.tools import tool                 # Decorador
from dotenv import load_dotenv                   # Função específica
import os                                        # Módulo inteiro
import json                                      # Módulo inteiro

Tipos de Import

Sintaxe Descrição Quando Usar Exemplo de Uso
import modulo Importa o módulo inteiro Várias funções do módulo os.getcwd()
from modulo import funcao Importa função específica Poucas funções específicas load_dotenv()

import modulo as alias Importa com apelido Nomes longos ou conflitos pd.DataFrame()

Variáveis e Tipos de Dados

Tipagem Dinâmica vs Estática

Uma das maiores diferenças entre Python e Java é como lidam com tipos de dados.

☕ Java - Tipagem Estática

// Java - DEVE declarar o tipo
String nome = "Wesley";
int idade = 25;
double temperatura = 36.5;
boolean ativo = true;
List lista = new ArrayList<>();
Map mapa = new HashMap<>();

// Erro se tentar mudar tipo:
// idade = "vinte e cinco";  // ERRO!
  • ✅ Detecta erros em tempo de compilação
  • ✅ Performance otimizada
  • ✅ IDE oferece melhor autocomplete
  • ❌ Mais verboso
  • ❌ Menos flexível

🐍 Python - Tipagem Dinâmica

# Python - tipo inferido automaticamente
nome = "Wesley"                    # str
idade = 25                         # int
temperatura = 36.5                 # float
ativo = True                       # bool
lista = ["a", "b", "c"]           # list
dicionario = {"key": "value"}      # dict

# Pode mudar tipo durante execução:
idade = "vinte e cinco"            # Agora é str!
  • ✅ Código mais conciso
  • ✅ Muito flexível
  • ✅ Prototipagem rápida
  • ❌ Erros só em tempo de execução
  • ❌ Pode ser confuso em projetos grandes

def processar_input(user_input):
    """Processa entrada do usuário de forma flexível."""
    if isinstance(user_input, str):
        return user_input.strip().lower()
    elif isinstance(user_input, (int, float)):
        return str(user_input)
    elif isinstance(user_input, dict):
        return json.dumps(user_input)
    else:
        return str(user_input)

Type Hints - O Melhor dos Dois Mundos

Python 3.5+ permite adicionar dicas de tipo, combinando flexibilidade com clareza!

Por que usar Type Hints?

  • Documentação: Código autodocumentado
  • IDE: Melhor autocomplete e detecção de erros
  • Manutenção: Mais fácil entender código antigo
  • Ferramentas: mypy pode verificar tipos
  • Colaboração: Outros devs entendem melhor

def processar_dados(
    dados: List[Dict[str, Union[str, int]]], 
    filtro: Optional[str] = None
) -> Tuple[List[Dict], int]:
    """
    Processa uma lista de dicionários.
    
    Args:

        filtro: Filtro opcional para aplicar

        Tupla com (dados_processados, quantidade)

    if filtro:
        dados_filtrados = [d for d in dados if filtro in str(d)]
    else:
        dados_filtrados = dados
    
    return dados_filtrados, len(dados_filtrados)

Dicionários - A Estrutura Fundamental

Por que Dicionários são TÃO Importantes?

Dicionários são EXTREMAMENTE usados em Python e no projeto LangChain! Eles são como HashMap em Java, mas muito mais poderosos e flexíveis.

Onde você vê dicionários no projeto:

  • Configurações: {"temperature": 0.7, "model": "gpt-4"}
  • Dados de entrada: {"input": "Olá", "session_id": "123"}
  • Respostas de API: {"content": "Resposta", "tokens": 50}
  • Metadados: {"source": "doc.pdf", "page": 1}
  • Session store: session_store[session_id] = history

# Dicionário de exemplo
config = {
    "model": "gpt-4",
    "temperature": 0.7,
    "max_tokens": 1000,
    "stream": False
}

# 1. KEYS() - Todas as chaves
chaves = config.keys()
print(list(chaves))  # ['model', 'temperature', 'max_tokens', 'stream']

# 2. VALUES() - Todos os valores
valores = config.values()
print(list(valores))  # ['gpt-4', 0.7, 1000, False]

# 3. ITEMS() - Pares chave-valor (MUITO USADO!)
for chave, valor in config.items():
    print(f"{chave}: {valor}")
# model: gpt-4
# temperature: 0.7
# max_tokens: 1000
# stream: False

# 4. GET() - Acesso seguro (JÁ VIMOS)
model = config.get("model", "gpt-3.5-turbo")  # Valor padrão
timeout = config.get("timeout", 30)           # 30 se não existir

# 5. SETDEFAULT() - Define valor se não existir
config.setdefault("timeout", 30)      # Só adiciona se não existir
config.setdefault("model", "gpt-3.5") # Não muda, já existe

print(config)  # Agora tem 'timeout': 30

# 6. POP() - Remove e retorna valor
stream_value = config.pop("stream")        # Remove e retorna False
timeout = config.pop("timeout", 60)       # Remove ou retorna padrão

# 7. POPITEM() - Remove último item (Python 3.7+)
ultimo_item = config.popitem()  # Remove e retorna ('max_tokens', 1000)

# 8. CLEAR() - Remove tudo
backup = config.copy()  # Faz cópia antes
config.clear()          # Agora está vazio: {}
config = backup         # Restaura

# 9. COPY() - Cópia superficial
config_copy = config.copy()
config_copy["model"] = "gpt-3.5"  # Não afeta o original

# 10. UPDATE() - Mescla dicionários (JÁ VIMOS)
novos_params = {"temperature": 0.9, "top_p": 0.95}
config.update(novos_params)

# 11. IN / NOT IN - Verificação de existência (MUITO USADO!)
if "model" in config:
    print(f"Modelo: {config['model']}")

if "api_key" not in config:
    config["api_key"] = os.getenv("OPENAI_API_KEY")

# 12. LEN() - Quantidade de itens
print(f"Configuração tem {len(config)} parâmetros")

# EXEMPLO PRÁTICO DO PROJETO
def configurar_llm(**kwargs):
    """Configura LLM com parâmetros flexíveis."""
    # Configuração padrão
    default_config = {
        "model": "gpt-3.5-turbo",
        "temperature": 0.7,
        "max_tokens": 1000,
        "timeout": 30
    }
    
    # Mescla com parâmetros fornecidos
    final_config = default_config.copy()
    final_config.update(kwargs)
    
    # Validação
    if "api_key" not in final_config:
        final_config["api_key"] = os.getenv("OPENAI_API_KEY")
    
    # Log da configuração
    print("Configuração do LLM:")
    for key, value in final_config.items():
        if key != "api_key":  # Não mostra a chave
            print(f"  {key}: {value}")
    
    return final_config

# Uso
config1 = configurar_llm()

config3 = configurar_llm(model="gpt-4", max_tokens=2000, stream=True)

# 3. PROCESSAMENTO DE ENTRADA DO USUÁRIO
def process_user_input(user_data: Dict[str, Any]) -> Dict[str, Any]:
    """Processa entrada do usuário."""
    # Validação de campos obrigatórios
    required_fields = ["input", "session_id"]
    for field in required_fields:
        if field not in user_data:
            raise ValueError(f"Campo obrigatório '{field}' não fornecido")
    
    # Configuração padrão
    processed_data = {
        "input": user_data["input"].strip(),
        "session_id": user_data["session_id"],
        "timestamp": datetime.now().isoformat(),
        "metadata": {}
    }
    
    # Adiciona metadados opcionais
    optional_fields = ["user_id", "context", "preferences"]
    for field in optional_fields:
        if field in user_data:
            processed_data["metadata"][field] = user_data[field]
    
    return processed_data

            "function": search_tool,
            "return_direct": False,
            "timeout": 30
        }
    ]
    
    # Processa configuração de cada tool
    processed_tools = []
    for tool_config in tools_config:
        # Validação
        if "name" not in tool_config or "function" not in tool_config:
            continue
        
        # Configuração padrão para tool
        tool_data = {
            "name": tool_config["name"],
            "description": tool_config.get("description", ""),
            "function": tool_config["function"],
            "return_direct": tool_config.get("return_direct", False),
            "timeout": tool_config.get("timeout", 60)
        }
        
        processed_tools.append(tool_data)
    
    return processed_tools

def format_response(content: str, **metadata) -> Dict[str, Any]:
    """Formata resposta de forma estruturada."""
    response = {
        "content": content,
        "timestamp": datetime.now().isoformat(),
        "status": "success",
        "metadata": {}
    }
    
    # Adiciona metadados fornecidos
    response["metadata"].update(metadata)
    
    # Adiciona informações automáticas
    response["metadata"].update({
        "content_length": len(content),
        "word_count": len(content.split()),
        "response_id": str(uuid.uuid4())
    })
    
    return response
